{
    char TSP[N+1] = "2104101279\0";  // TSP问题的数据
    double A[N], B[N]; 
    int i = 0; 
    for(i; i < N; i++)
    {
        exe_type("a.out", &A[i], &TSP[i]); // A.exe 执行, 解决 TSP[i] 问题并将结果输出

    }
    return 0;
}
